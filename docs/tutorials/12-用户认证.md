## 12.2 UserService

```typescript
// src/services/UserService.ts

export class UserService {
  /** 登录 */
  static async login(account: string, password: string): Promise<User> {
    const body: LoginRequest = { userAccount: account, userPassword: password }
    const response = await httpUtil.post<User>(API_CONFIG.PATHS.USER_LOGIN, body)
    // 登录成功后保存 Token
    if (response.code === 0 && response.data?.token) {
      UniStorage.setToken(response.data.token)
    }
    return response.data
  }

  /** 注册 */
  static async register(account: string, password: string, checkPassword: string): Promise<User> {
    const body: RegisterRequest = { userAccount: account, userPassword: password, checkPassword }
    const response = await httpUtil.post<User>(API_CONFIG.PATHS.USER_REGISTER, body)
    if (response.code === 0 && response.data?.token) {
      UniStorage.setToken(response.data.token)
    }
    return response.data
  }

  /** 获取当前用户信息 */
  static async getUserInfo(): Promise<User> {
    const response = await httpUtil.get<User>(API_CONFIG.PATHS.USER_INFO)
    return response.data
  }

  /** 登出 */
  static async logout(): Promise<void> {
    try {
      await httpUtil.post(API_CONFIG.PATHS.USER_LOGOUT)
    } finally {
      UniStorage.clearToken()
    }
  }
}
```

注意：Token 在 Service 层保存，在 Store 层管理登录状态。两层职责分明。

## 12.4 useAuth 组合式函数

`src/hooks/useAuth.ts` 封装认证相关的业务逻辑：

```typescript
export function useAuth() {
  const userStore = useUserStore()

  const isLoggedIn = computed(() => userStore.isLogin)
  const userName = computed(() => userStore.userName)
  const userAvatar = computed(() => userStore.userAvatar)

  /** 检查登录状态，未登录则跳转 */
  function requireAuth(): boolean {
    if (!userStore.isLogin) {
      uni.showToast({ title: '请先登录', icon: 'none' })
      uni.navigateTo({ url: '/pages/mine/login' })
      return false
    }
    return true
  }

  async function login(account: string, password: string) {
    return userStore.login(account, password)
  }

  async function logout() {
    await userStore.logout()
  }

  return { isLoggedIn, userName, userAvatar, requireAuth, login, logout }
}
```

使用示例：
```typescript
const { requireAuth } = useAuth()

function onOrderClick() {
  if (!requireAuth()) return  // 未登录自动跳转登录页
  uni.navigateTo({ url: '/pages/order/order-preview' })
}
```

## 12.5 全局路由守卫

`src/main.ts` 中通过 mixin 实现页面级鉴权：

```typescript
const authRequiredPages = [
  'pages/order/order-preview',
  'pages/order/order-detail',
  'pages/order/payment',
  'pages/address/add-address',
  'pages/review/review-submit'
]

app.mixin({
  onLoad() {
    const pages = getCurrentPages()
    const currentPage = pages[pages.length - 1]
    const route = currentPage?.route || ''
    if (authRequiredPages.includes(route)) {
      const token = uni.getStorageSync('token')
      if (!token) {
        uni.redirectTo({ url: '/pages/mine/login' })
      }
    }
  }
})
```

这是 uni-app 中实现路由守卫的常用方式：在每个页面的 `onLoad` 生命周期中检查权限。

## 12.6 Token 管理

Token 存储在两个地方，保持同步：

1. **UniStorage**：`uni.setStorageSync('token', token)` — 用于 Axios 拦截器读取
2. **UserStore persist**：Pinia 持久化 `token` 和 `isLogin` — 用于 UI 状态判断

```typescript
// src/utils/StorageUtil.ts
export class UniStorage {
  static getToken(): string {
    return uni.getStorageSync('token') || ''
  }
  static setToken(token: string): void {
    uni.setStorageSync('token', token)
  }
  static clearToken(): void {
    uni.removeStorageSync('token')
  }
}
```
