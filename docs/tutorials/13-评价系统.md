## 13.2 ReviewService

```typescript
// src/services/ReviewService.ts

export class ReviewService {
  /** 分页获取评价 */
  static async getReviews(shopId: string, pageNum: number = 1, pageSize: number = 10): Promise<ReviewPageResponse> {
    const params: Record<string, unknown> = { shopId, pageNum, pageSize }
    const response = await httpUtil.get<ReviewPageResponse>(API_CONFIG.PATHS.REVIEW_LIST, params)
    return response.data || { list: [], total: 0, page: 1, pageSize: 10 }
  }

  /** 提交评价 */
  static async submitReview(request: ReviewSubmitRequest): Promise<Review> {
    const response = await httpUtil.post<Review>(API_CONFIG.PATHS.REVIEW_SUBMIT, request)
    return response.data
  }

  /** 切换点赞状态 */
  static async toggleLike(reviewId: string): Promise<Review> {
    const url = API_CONFIG.PATHS.REVIEW_TOGGLE_LIKE.replace('{reviewId}', reviewId)
    const response = await httpUtil.post<Review>(url)
    return response.data
  }
}
```

### 敏感词过滤

提交前使用 `SensitiveFilter` 验证内容：

```typescript
// src/utils/SensitiveFilter.ts
export class SensitiveFilter {
  private static readonly SENSITIVE_WORDS = ['垃圾', '难吃', '恶心', '骗子', ...]

  static validateContent(content: string): { valid: boolean; message: string } {
    if (!content || content.trim().length === 0) {
      return { valid: false, message: '评价内容不能为空' }
    }
    if (content.trim().length < 5) {
      return { valid: false, message: '评价内容不能少于5个字' }
    }
    if (content.trim().length > 500) {
      return { valid: false, message: '评价内容不能超过500字' }
    }
    if (SensitiveFilter.hasSensitiveWord(content)) {
      const words = SensitiveFilter.getSensitiveWords(content)
      return { valid: false, message: `评价内容包含敏感词：${words.join('、')}` }
    }
    return { valid: true, message: '' }
  }
}
```

### 提交评价

```typescript
async function onSubmit() {
  const validation = SensitiveFilter.validateContent(content.value)
  if (!validation.valid) {
    ToastUtil.showMessage(validation.message)
    return
  }

  await ReviewService.submitReview({
    orderId: orderId.value,
    foodId: foodId.value,
    shopId: shopId.value,
    rating: rating.value,
    content: content.value,
    images: images.value,
    tags: selectedTags.value,
    isAnonymous: isAnonymous.value
  })

  ToastUtil.showSuccess('评价提交成功')
  uni.navigateBack()
}
```
