# 第四章：状态管理（Pinia）

## 4.1 Pinia 安装与初始化

项目使用 Pinia 替代 Vuex 作为状态管理方案。初始化在 `src/main.ts` 中：

```typescript
import { createSSRApp } from 'vue'
import { createPinia } from 'pinia'
import { createPersistedState } from 'pinia-plugin-persistedstate'

export function createApp() {
  const app = createSSRApp(App)
  const pinia = createPinia()

  // 配置持久化插件，使用 uni-app 的存储 API
  pinia.use(createPersistedState({
    storage: {
      getItem: (key: string) => uni.getStorageSync(key),
      setItem: (key: string, value: string) => uni.setStorageSync(key, value),
    }
  }))

  app.use(pinia)

  return { app, pinia }
}
```

关键点：uni-app 不能使用 `localStorage`，必须适配为 `uni.getStorageSync` / `uni.setStorageSync`。

## 4.2 用户状态管理

`src/stores/user.store.ts` 管理用户登录状态：

```typescript
export const useUserStore = defineStore('user', () => {
  // State
  const token = ref<string>(UniStorage.getToken())
  const isLogin = ref<boolean>(token.value.length > 0)
  const userInfo = ref<User | null>(null)

  // Getters
  const userName = computed(() => userInfo.value?.userName || '未登录')
  const userAvatar = computed(() => userInfo.value?.userAvatar || '')

  // Actions
  async function login(account: string, password: string): Promise<boolean> {
    try {
      ToastUtil.showLoading('登录中...')
      const user = await UserService.login(account, password)
      userInfo.value = user
      token.value = user.token || UniStorage.getToken()
      isLogin.value = true
      ToastUtil.hideLoading()
      ToastUtil.showSuccess('登录成功')
      return true
    } catch (e) {
      ToastUtil.hideLoading()
      return false
    }
  }

  async function logout(): Promise<void> {
    try {
      await UserService.logout()
    } finally {
      token.value = ''
      isLogin.value = false
      userInfo.value = null
      UniStorage.clearToken()
    }
  }

  return { token, isLogin, userInfo, userName, userAvatar, login, logout, fetchUserInfo, register }
}, {
  persist: {
    key: 'user-store',
    storage: {
      getItem: (key) => uni.getStorageSync(key),
      setItem: (key, value) => uni.setStorageSync(key, value)
    },
    paths: ['token', 'isLogin']  // 只持久化 token 和 isLogin
  }
})
```

设计要点：
- **Composition API 风格**：使用 `ref`、`computed` 定义状态
- **选择性持久化**：只持久化 `token` 和 `isLogin`，`userInfo` 每次从后端获取
- **Token 双重存储**：Pinia 持久化 + `UniStorage` 工具类，确保一致性

## 4.3 购物车状态管理

`src/stores/cart.store.ts` 管理购物车数据，**不持久化**（刷新丢失）：

```typescript
export const useCartStore = defineStore('cart', () => {
  const choosePrice = ref<number>(0)
  const buyFoods = ref<Record<string, CartFoodItem>>({})

  const totalItems = computed(() => Object.keys(buyFoods.value).length)
  const totalPrice = computed(() => MathUtil.toFixed(choosePrice.value, 2))
  const buyFoodList = computed(() => Object.values(buyFoods.value))

  function addToCart(item: CartFoodItem): void {
    const existingItem = buyFoods.value[item.id]
    if (existingItem) {
      existingItem.buyNum += item.buyNum || 1
      choosePrice.value = MathUtil.addition(choosePrice.value, MathUtil.multiply(item.price, item.buyNum || 1))
    } else {
      const newItem = { ...item, buyNum: item.buyNum || 1 }
      buyFoods.value[item.id] = newItem
      choosePrice.value = MathUtil.addition(choosePrice.value, MathUtil.multiply(item.price, newItem.buyNum))
    }
  }

  function decreaseQuantity(foodId: string): void {
    const item = buyFoods.value[foodId]
    if (item) {
      if (item.buyNum > 1) {
        item.buyNum -= 1
        choosePrice.value = MathUtil.subtraction(choosePrice.value, item.price)
      } else {
        removeFromCart(foodId)  // 数量为 0 时自动移除
      }
    }
  }

  return { buyFoods, totalItems, totalPrice, buyFoodList, addToCart, increaseQuantity, decreaseQuantity, removeFromCart, clearCart }
})
```

购物车用 `Record<string, CartFoodItem>` 以食品 ID 为 key，方便快速查找和去重。价格计算使用 `MathUtil` 避免浮点精度问题。

## 4.4 订单状态管理

`src/stores/order.store.ts` 管理订单数据：

```typescript
export const useOrderStore = defineStore('order', () => {
  const currentOrder = ref<Order | null>(null)
  const orders = ref<Order[]>([])

  async function fetchOrders(page: number = 1, pageSize: number = 10, status?: OrderStatus): Promise<void> {
    const response = await OrderService.getOrders(page, pageSize, status)
    if (page === 1) {
      orders.value = response.list || []
    } else {
      orders.value = [...orders.value, ...(response.list || [])]  // 追加分页数据
    }
  }

  async function cancelOrder(orderId: string): Promise<void> {
    await OrderService.cancelOrder(orderId)
    // 同步更新本地状态
    if (currentOrder.value?.id === orderId) {
      currentOrder.value.status = OrderStatus.CANCELLED
    }
    const idx = orders.value.findIndex(o => o.id === orderId)
    if (idx !== -1) orders.value[idx].status = OrderStatus.CANCELLED
  }

  return { currentOrder, orders, fetchOrders, fetchOrderDetail, cancelOrder, completeOrder }
})
```

订单状态变更时，先调用后端 API，再同步更新本地状态，保持 UI 即时响应。

## 4.5 店铺状态管理

export const useShopStore = defineStore('shop', () => {
  const currentShop = ref<Shop | null>(null)
  const shops = ref<Shop[]>([])

  async function fetchShops(): Promise<void> {
    shops.value = await ShopService.getShops()
  }

  async function fetchShopDetail(shopId: string): Promise<void> {
    currentShop.value = await ShopService.getShopDetail(shopId)

})

## 4.6 全局应用状态

`src/stores/app.store.ts` 存储全局 UI 状态：

```typescript
export const useAppStore = defineStore('app', () => {
  const selectedIndex = ref<number>(0)      // 当前 Tab 索引
  const topRectHeight = ref<number>(0)       // 状态栏高度
  const bottomRectHeight = ref<number>(0)    // 底部栏高度

  function setSelectedIndex(index: number) {
    selectedIndex.value = index
  }

  return { selectedIndex, topRectHeight, bottomRectHeight, setSelectedIndex }
})
```

## 小结

本章讲解了 5 个 Pinia Store 的设计。核心模式是：**Composition API 风格 + 选择性持久化 + Service 调用**。Store 只负责状态管理，业务逻辑委托给 Service。下一章我们将讲解类型系统设计。
